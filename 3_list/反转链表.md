class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(head == nullptr)
            return nullptr;
       ListNode* curNode = head;
       ListNode* curPrev = nullptr;
       ListNode* curNext = head->next;
       if(curNext == NULL)
            return head;
        while(curNode){
            curNode->next = curPrev;
            curPrev = curNode;
            curNode = curNext;
            if(curNext)
                curNext = curNext->next;
        }
        return curPrev;
    }
};
