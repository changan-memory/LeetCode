



# 我的解法

```cpp
class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        if(head->next == NULL)
            return head;

        ListNode* curNode = head;
        ListNode* midNode = nullptr;
        int count = 0;       
        while(curNode){
            count++;
            curNode = curNode->next;
            if(count == 2){
                midNode = head;
                continue;
            }
            if(count % 2 ==0){
                midNode = midNode->next;
            }
        }
        return midNode->next;
    }
};
```



# 快慢指针解法

```cpp
class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        ListNode* slow = head;
        ListNode* fast = head;
        while (fast && fast->next){
            slow = slow->next;
            fast = fast->next->next;
        }
        return slow;
    }
};
```
